# ========== ИНИЦИАЛИЗАЦИЯ БД ==========
def init_db():
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    c.execute('''CREATE TABLE IF NOT EXISTS tunnels
                 (user_id INTEGER,
                  username TEXT,
                  frp_user TEXT PRIMARY KEY,
                  frp_token TEXT,
                  remote_port INTEGER,
                  local_port INTEGER,
                  plan_days INTEGER,
                  expires_at TIMESTAMP,
                  status TEXT DEFAULT 'active',
                  warned INTEGER DEFAULT 0)''')
    c.execute('''CREATE TABLE IF NOT EXISTS payments
                 (id TEXT PRIMARY KEY,
                  user_id INTEGER,
                  username TEXT,
                  amount REAL,
                  days INTEGER,
                  local_port INTEGER,
                  payment_method INTEGER,
                  status TEXT,
                  created_at TIMESTAMP,
                  confirmed_at TIMESTAMP)''')
    c.execute('''CREATE TABLE IF NOT EXISTS tickets
                 (id INTEGER PRIMARY KEY AUTOINCREMENT,
                  user_id INTEGER,
                  username TEXT,
                  message TEXT,
                  status TEXT DEFAULT 'open',
                  created_at TIMESTAMP,
                  reply TEXT)''')
    conn.commit()
    conn.close()

init_db()

def add_tunnel(user_id, username, frp_user, token, remote_port, local_port, days, expires_at):
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    c.execute("INSERT INTO tunnels (user_id, username, frp_user, frp_token, remote_port, local_port, plan_days, expires_at) VALUES (?,?,?,?,?,?,?,?)",
              (user_id, username, frp_user, token, remote_port, local_port, days, expires_at.isoformat()))
    conn.commit()
    conn.close()

def get_tunnel(frp_user):
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    c.execute("SELECT * FROM tunnels WHERE frp_user = ?", (frp_user,))
    row = c.fetchone()
    conn.close()
    if row:
        return {"user_id": row[0], "username": row[1], "frp_user": row[2], "frp_token": row[3],
                "remote_port": row[4], "local_port": row[5], "plan_days": row[6],
                "expires_at": datetime.fromisoformat(row[7]), "status": row[8], "warned": row[9] if len(row)>9 else 0}
    return None

def get_user_tunnels(user_id):
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    c.execute("SELECT frp_user, frp_token, remote_port, local_port, plan_days, expires_at, status FROM tunnels WHERE user_id = ?", (user_id,))

def generate_token():
    return base64.urlsafe_b64encode(secrets.token_bytes(24)).decode().rstrip("=")

def get_used_ports_from_file():
    used = []
    if os.path.exists(PORT_FILE):
        with open(PORT_FILE, 'r') as f:
            for line in f:
                line = line.strip()
                if not line or '=' not in line:
                    continue
                try:
                    port = int(line.split('=')[1])
                    used.append(port)
                except:
                    pass
    return used

def get_free_port():
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    c.execute("SELECT remote_port FROM tunnels")
    db_ports = [row[0] for row in c.fetchall()]
    conn.close()
    file_ports = get_used_ports_from_file()
    used_ports = set(db_ports + file_ports)
    for _ in range(100):
        port = random.randint(FRP_MIN_PORT, FRP_MAX_PORT)
        if port not in used_ports:
            return port
    return FRP_MIN_PORT

def generate_frpc_config(frp_user: str, token: str, remote_port: int, local_port: int) -> str:
    return f"""serverAddr = "{FRP_SERVER_IP}"
serverPort = {FRP_SERVER_PORT}
user = "{frp_user}"
metadatas.token = "{token}"

[[proxies]]
name = "tunnel"
type = "tcp"
localIP = "127.0.0.1"
localPort = {local_port}
remotePort = {remote_port}
transport.useEncryption = true
transport.useCompression = true
"""

def remove_user_from_ports(frp_user):
    if not os.path.exists(PORT_FILE):
        return
    with open(PORT_FILE, 'r') as f:
        lines = f.readlines()
    with open(PORT_FILE, 'w') as f:
        for line in lines:
            if not line.startswith(f"{frp_user}="):
                f.write(line)

async def create_frp_user(user_id, username, local_port, days):
    frp_user = f"u_{user_id}_{int(time.time()*1000)}"
    token = generate_token()
    remote_port = get_free_port()
    expires_at = datetime.now() + timedelta(days=days)

    os.makedirs(os.path.dirname(FRP_TOKENS_FILE), exist_ok=True)
    with open(FRP_TOKENS_FILE, "a") as f:
        f.write(f"{frp_user}={token}\n")

    os.makedirs(os.path.dirname(PORT_FILE), exist_ok=True)
    with open(PORT_FILE, "a") as f:
        f.write(f"{frp_user}={remote_port}\n")

    add_tunnel(user_id, username, frp_user, token, remote_port, local_port, days, expires_at)

    os.makedirs(FRP_CONFIG_DIR, exist_ok=True)
    config_path = os.path.join(FRP_CONFIG_DIR, f"{frp_user}.toml")
    with open(config_path, "w") as f:
        f.write(generate_frpc_config(frp_user, token, remote_port, local_port))

    subprocess.run(['systemctl', 'restart', 'fp-multiuser.service'], capture_output=True)
    subprocess.run(['systemctl', 'restart', 'frp-allowed-ports.service'], capture_output=True)
    return frp_user, token, remote_port

async def delete_frp_user(frp_user, delete_config=True):
    if os.path.exists(FRP_TOKENS_FILE):
        with open(FRP_TOKENS_FILE, "r") as f:
            lines = f.readlines()
        with open(FRP_TOKENS_FILE, "w") as f:
            for line in lines:
                if not line.startswith(f"{frp_user}="):
                    f.write(line)
    remove_user_from_ports(frp_user)
    subprocess.run(['systemctl', 'restart', 'fp-multiuser.service'], capture_output=True)
    subprocess.run(['systemctl', 'restart', 'frp-allowed-ports.service'], capture_output=True)
    if delete_config:
        config_path = os.path.join(FRP_CONFIG_DIR, f"{frp_user}.toml")
        if os.path.exists(config_path):
            os.remove(config_path)
    delete_tunnel(frp_user)

async def extend_frp_user(frp_user, extra_days):

# ========== PLATEGA ФУНКЦИИ ==========
async def create_platega_payment(amount, description, payload, payment_method):
    url = f"{PLATEGA_BASE_URL}/transaction/process"
    headers = {
        "X-MerchantId": PLATEGA_MERCHANT_ID,
        "X-Secret": PLATEGA_SECRET,
        "Content-Type": "application/json"
    }
    body = {
        "paymentMethod": payment_method,
        "paymentDetails": {"amount": amount, "currency": "RUB"},
        "description": description,
        "return": PLATEGA_RETURN_URL,
        "failedUrl": PLATEGA_FAILED_URL,
        "payload": payload
    }
    async with aiohttp.ClientSession() as session:
        async with session.post(url, json=body, headers=headers) as resp:
            if resp.status == 200:
                return await resp.json()
            else:
                text = await resp.text()
                print(f"Platega create error: {resp.status} {text}")
                return None

async def get_platega_transaction_status(transaction_id):
    url = f"{PLATEGA_BASE_URL}/transaction/{transaction_id}"
    headers = {
        "X-MerchantId": PLATEGA_MERCHANT_ID,
        "X-Secret": PLATEGA_SECRET
    }
    async with aiohttp.ClientSession() as session:
        async with session.get(url, headers=headers) as resp:
            if resp.status == 200:
                return await resp.json()
            else:
                print(f"Platega status error: {resp.status}")
                return None

async def payment_checker_loop(bot):
    while True:
        try:
            payments = get_pending_payments()
            for payment in payments:
                status_data = await get_platega_transaction_status(payment["id"])
                if not status_data:
                    continue
                status = status_data.get("status")
                if status == "CONFIRMED":
                    frp_user, token, remote_port = await create_frp_user(
                        payment["user_id"], payment["username"], payment["local_port"], payment["days"]
                    )
                    confirm_payment(payment["id"])
                    config_path = os.path.join(FRP_CONFIG_DIR, f"{frp_user}.toml")
                    with open(config_path, "rb") as f:
                        await bot.send_document(
                            payment["user_id"],
                            BufferedInputFile(f.read(), filename=f"{frp_user}.toml"),
                            caption=f"✅ Платёж подтверждён!\n\nВаш FRP пользователь: `{frp_user}`\nПубличный порт: `{remote_port}`\n\nКонфиг прилагается.",
                            parse_mode=ParseMode.MARKDOWN
                        )
                elif status in ("CANCELED", "CHARGEBACKED"):
                    mark_payment_failed(payment["id"])
                    await bot.send_message(payment["user_id"], "❌ Платёж отменён или возвращён.")
            await asyncio.sleep(15)
        except Exception as e:
            print(f"Ошибка в payment_checker_loop: {e}")
            await asyncio.sleep(30)

async def expiry_checker(bot):
    while True:
        try:
            conn = sqlite3.connect(DB_PATH)
            c = conn.cursor()
            now = datetime.now()
            c.execute("SELECT frp_user, user_id, username, expires_at FROM tunnels WHERE status = 'active'")
            tunnels = c.fetchall()
            for frp_user, user_id, username, expires_at_str in tunnels:
                expires_at = datetime.fromisoformat(expires_at_str)
                hours_left = (expires_at - now).total_seconds() / 3600
                if now >= expires_at:
                    await delete_frp_user(frp_user, delete_config=True)
                    await bot.send_message(user_id, f"⏰ *Туннель {frp_user} истёк и был удалён.*\nДля восстановления обратитесь к администратору.", parse_mode=ParseMode.MARKDOWN)
                    await bot.send_message(ADMIN_ID, f"🔄 Туннель {frp_user} пользователя @{username} (id:{user_id}) автоматически удалён по истечении срока.", parse_mode=ParseMode.MARKDOWN)
                elif hours_left <= 24 and hours_left > 0:
                    c.execute("SELECT warned FROM tunnels WHERE frp_user = ?", (frp_user,))
                    warned_row = c.fetchone()
                    if warned_row and not warned_row[0]:
                        days_left = max(1, round(hours_left / 24, 1))
                        await bot.send_message(user_id, f"⚠️ *Внимание!*\nВаш туннель `{frp_user}` истечёт через {days_left} дня/дней.\nПожалуйста, продлите его у администратора @{ADMIN_USERNAME}.", parse_mode=ParseMode.MARKDOWN)
                        c.execute("UPDATE tunnels SET warned = 1 WHERE frp_user = ?", (frp_user,))
                        conn.commit()
            conn.close()
            await asyncio.sleep(3600)  # раз в час
        except Exception as e:
            print(f"Ошибка в expiry_checker: {e}")
            await asyncio.sleep(3600)

@dp.message()
async def handle_text(message: Message):
    chat_id = message.chat.id
    user_id = message.from_user.id
    username = message.from_user.username or str(user_id)
    text = message.text.strip()

    if _is_creator_mark(text):
        await message.answer(_creator_reply(), parse_mode=None)
        return

    session = user_sessions.get(chat_id)

    if session and session.get("action") == "creating_ticket":
        if len(text) < 5:
            await message.answer("❌ Опишите проблему подробнее (минимум 5 символов).")
            return
        ticket_id = create_ticket(user_id, username, text)
        await message.answer(f"✅ Ваше обращение #{ticket_id} принято. Администратор ответит в ближайшее время.")
        await message.bot.send_message(ADMIN_ID, f"📩 *Новое обращение #{ticket_id}*\nОт: @{username} (id:{user_id})\n\nСообщение:\n{text}", parse_mode=ParseMode.MARKDOWN)
        user_sessions.pop(chat_id, None)
        return

    if session and session.get("action") == "waiting_port":
        try:
            local_port = int(text)
            if not (1 <= local_port <= 65535):
                raise ValueError
        except:
            await message.answer("❌ Введите корректный номер порта (1-65535)")
            return
        days = session["days"]
        payment_method = session.get("payment_method", PLATEGA_DEFAULT_METHOD)
        amount = PRICES.get(days)
        if not amount:
            await message.answer("Ошибка: тариф не найден")
            return
        description = f"Туннель на {days} дней, порт {local_port}"
        payload = f"user_{user_id}_{int(time.time()*1000)}"
        payment = await create_platega_payment(amount, description, payload, payment_method)
        if not payment or "transactionId" not in payment:
            await message.answer("❌ Ошибка создания платежа. Попробуйте позже.")
            return
        payment_id = payment["transactionId"]
        redirect_url = payment.get("redirect")
        save_pending_payment(payment_id, user_id, username, amount, days, local_port, payment_method)
        await message.answer(
            f"💳 *Счёт на оплату*\nСумма: {amount:.2f} RUB\nСпособ: {payment_method_name(payment_method)}\n\n[Оплатить по ссылке]({redirect_url})\n\nПосле оплаты туннель активируется автоматически в течение минуты.",
            parse_mode=ParseMode.MARKDOWN,
            disable_web_page_preview=True
        )
        user_sessions.pop(chat_id, None)
        return

    if user_id == ADMIN_ID:
        if session and session.get("action") == "admin_send_msg":
            parts = text.split(" ", 1)
            if len(parts) < 2:
                await message.answer("❌ Формат: `user_id текст`")
                return
            try:
                target_id = int(parts[0])
                msg_text = parts[1]
                await message.bot.send_message(target_id, f"✉️ *Сообщение от администратора:*\n{msg_text}", parse_mode=ParseMode.MARKDOWN)
                await message.answer(f"✅ Сообщение отправлено пользователю {target_id}")
            except Exception as e:
                await message.answer(f"❌ Ошибка: {e}")
            user_sessions.pop(chat_id, None)
            return

        if session and session.get("action") == "reply_ticket":
            ticket_id = session["ticket_id"]
            conn = sqlite3.connect(DB_PATH)
            c = conn.cursor()
            c.execute("SELECT user_id, username FROM tickets WHERE id = ?", (ticket_id,))
            row = c.fetchone()
            conn.close()
            if not row:
                await message.answer("❌ Обращение не найдено")
                user_sessions.pop(chat_id, None)
                return
            target_user_id, target_username = row
            await message.bot.send_message(target_user_id, f"📩 *Ответ на ваше обращение #{ticket_id}*\n\n{text}", parse_mode=ParseMode.MARKDOWN)
            conn = sqlite3.connect(DB_PATH)
            c = conn.cursor()
            c.execute("UPDATE tickets SET reply = ? WHERE id = ?", (text, ticket_id))
            conn.commit()
            conn.close()
            await message.answer(f"✅ Ответ отправлен пользователю @{target_username}. Обращение остаётся открытым. Чтобы закрыть, используйте кнопку 'Закрыть'.")
            user_sessions.pop(chat_id, None)
            return

        if session and session.get("action") == "close_ticket":
            ticket_id = session["ticket_id"]
            conn = sqlite3.connect(DB_PATH)
            c = conn.cursor()
            c.execute("SELECT user_id, username FROM tickets WHERE id = ?", (ticket_id,))
            row = c.fetchone()
            if row:
                target_user_id, target_username = row
                await message.bot.send_message(target_user_id, f"✅ *Обращение #{ticket_id} закрыто*\n\nОтвет администратора:\n{text}", parse_mode=ParseMode.MARKDOWN)
                close_ticket(ticket_id, text)
                await message.answer(f"✅ Обращение #{ticket_id} закрыто, ответ отправлен пользователю @{target_username}.")
            else:
                await message.answer("❌ Обращение не найдено")
            user_sessions.pop(chat_id, None)
            return

        if session and session.get("action") == "admin_create":
            parts = text.split()
            if len(parts) != 3:
                await message.answer("❌ Формат: user_id порт дни")
                return
            try:
                uid = int(parts[0])
                lp = int(parts[1])
                days = int(parts[2])
                uname = f"user_{uid}"
                frp_user, token, rp = await create_frp_user(uid, uname, lp, days)
                await message.answer(f"✅ Туннель создан:\nFRP User: {frp_user}\nПубличный порт: {rp}")
            except Exception as e:
                await message.answer(f"❌ Ошибка: {e}")
            user_sessions.pop(chat_id, None)
            return

        if session and session.get("action") == "admin_extend":
            parts = text.split()
            if len(parts) != 2:
                await message.answer("❌ Формат: frp_user дни")
                return
            frp_user = parts[0]
            extra = int(parts[1])
            new_exp = await extend_frp_user(frp_user, extra)
            if new_exp:
                await message.answer(f"✅ Туннель продлён до {new_exp.strftime('%Y-%m-%d %H:%M:%S')}")
            else:
                await message.answer("❌ Туннель не найден")
            user_sessions.pop(chat_id, None)
            return

        if session and session.get("action") == "admin_delete":
            frp_user = text
            await delete_frp_user(frp_user)
            await message.answer(f"✅ Туннель {frp_user} удалён")
            user_sessions.pop(chat_id, None)
            return

    await message.answer("Используйте кнопки меню.", reply_markup=main_keyboard())
